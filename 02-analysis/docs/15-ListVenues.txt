@startuml
title Venue Management Navigation

actor "User" as user
control ": AdminDashboard" as dashboard
boundary ": OrganizerVenueListPage" as organizerVenueList
boundary ": AdminVenueListPage" as adminVenueList

== 1. User clicks "Manage Venues"\n 2.The system navigates to the Venue List Page==

user -> dashboard: Click "Manage Venues"
activate dashboard

alt If the user is an Organizer
    dashboard -> organizerVenueList: Navigate to Organizer Venue List Page
    organizerVenueList --> user: Display Organizer Venue List Page
else If the user is an Administrator
    dashboard -> adminVenueList: Navigate to Admin Venue List Page
    adminVenueList --> user: Display Admin Venue List Page
end

deactivate dashboard
@enduml
