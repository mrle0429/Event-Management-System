@startuml
title Event Management Navigation

actor "User" as user
control ": AdminDashboard" as dashboard
boundary ": OrganizerEventListPage" as organizerEventList
boundary ": AdminEventListPage" as adminEventList

== 1. User clicks "Manage Events" \n2. The system navigates to the Event List Page==

user -> dashboard: Click "Manage Events"
activate dashboard

alt If the user is an Organizer
    dashboard -> organizerEventList: Navigate to Organizer Event List Page
    organizerEventList --> user: Display Organizer Event List Page
else If the user is an Administrator
    dashboard -> adminEventList: Navigate to Admin Event List Page
    adminEventList --> user: Display Admin Event List Page
end

deactivate dashboard
@enduml
